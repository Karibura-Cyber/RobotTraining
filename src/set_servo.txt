// ฟังก์ชั่นการเตรียม servo
void set_servo(){
  int deg = knob(0,180);                // ตัวแปร deg รับค่าจาก KNOB โดยกำหนดค่าตั้งแต่ 0 - 180
  oled.clear();                         // ล้างหน้าจอ
  oled.text(0, 0, "%d", deg);           // แสดงค่าที่ได้จากตัวแปร deg
  oled.show();                          // แสดงข้อความบนหน้าจอ
  servo(1, deg);                        // สั่ง servo ช่องที่ 1 ให้หมุน โดยค่าองศามาจากตัวแปร deg ข้างบน
}
